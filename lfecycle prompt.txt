markdown# TASK: Build Lifecycle Management Marketplace (Stage 1) + Stage 2 Service Hub

## OVERVIEW

Build a complete Lifecycle Management service from scratch, including both Stage 1 marketplace page and Stage 2 service hub pages. Lifecycle Management provides governance frameworks, stage-gate processes, project/product lifecycle templates, and compliance workflows for managing digital assets from inception through retirement. This service has a close integration relationship with Portfolio Management, where Portfolio identifies what needs to be done and Lifecycle provides the processes to execute.

## PART 5: CREATE DATA STRUCTURES

### Task: Create `lifecycleData.ts`

**Location:** Same directory as other data files

**IMPORTANT:** This file should import and reference Portfolio Management data for integration

**Data Structures Needed:**
```typescript
// Import Portfolio data for integration
import { applications, projects } from './portfolioData';

// Lifecycle Stage
interface LifecycleStage {
  id: string;
  name: string;
  description: string;
  order: number;
  durationEstimate: string; // e.g., "2-4 weeks"
  deliverables: string[];
  entryGate?: StageGate;
  exitGate: StageGate;
  activities: StageActivity[];
}

// Stage Gate (Approval Point)
interface StageGate {
  id: string;
  name: string;
  description: string;
  criteria: GateCriteria[];
  requiredApprovers: ApproverRole[];
  optionalApprovers: ApproverRole[];
  documents: string[]; // Required documentation
  estimatedReviewTime: string;
}

interface GateCriteria {
  id: string;
  criterion: string; // "Business case approved", "Risk assessment completed"
  mandatory: boolean;
  evidenceRequired: string; // What proves this is met
}

interface ApproverRole {
  role: string; // "Chief Architect", "Business Owner", "Security Lead"
  level: 'required' | 'optional' | 'informed';
}

// Stage Activity
interface StageActivity {
  id: string;
  name: string;
  description: string;
  owner: string; // Role responsible
  estimatedEffort: string;
  dependencies: string[]; // Activity IDs
  outputs: string[];
}

// Lifecycle Template
interface LifecycleTemplate {
  id: string; // Format: LCT-001
  title: string;
  templateType: 'project' | 'product' | 'application' | 'retirement';
  category: string;
  description: string;
  longDescription: string;
  
  // Metadata
  author: {
    name: string;
    role: string;
  };
  version: string;
  lastUpdated: string;
  
  // Lifecycle Definition
  methodology: 'waterfall' | 'agile' | 'hybrid' | 'stage-gate';
  totalDuration: string; // e.g., "12-16 weeks"
  stages: LifecycleStage[];
  
  // Governance
  governanceFramework: string;
  complianceRequirements: string[];
  riskLevel: 'low' | 'medium' | 'high';
  
  // Applicability
  suitableFor: string[];
  prerequisites: string[];
  
  // Documentation
  templates: TemplateDocument[];
  checklists: Checklist[];
  guidelines: string; // Markdown
  
  // Success Metrics
  successCriteria: string[];
  kpis: string[];
  
  // Examples
  caseStudies: {
    title: string;
    description: string;
    outcome: string;
    duration: string;
  }[];
  
  // Relationships
  relatedTemplates: string[];
  
  // Engagement
  usageCount: number;
  avgCompletionTime: string;
  successRate: number; // Percentage
  rating: number;
  reviewCount: number;
  
  // Tags
  tags: string[];
  complexity: 'simple' | 'moderate' | 'complex';
}

interface TemplateDocument {
  name: string;
  type: 'word' | 'excel' | 'ppt' | 'pdf' | 'markdown';
  purpose: string;
  url: string;
}

interface Checklist {
  id: string;
  name: string;
  stage: string; // Which stage this applies to
  items: ChecklistItem[];
}

interface ChecklistItem {
  id: string;
  description: string;
  required: boolean;
  owner: string; // Role responsible
}

// Active Lifecycle Instance (Project/Product/App in lifecycle)
interface LifecycleInstance {
  id: string; // Format: LCI-2026-001
  name: string;
  type: 'project' | 'product' | 'application-retirement' | 'application-modernization';
  basedOnTemplate: string; // Template ID
  
  // Link to Portfolio (CRITICAL INTEGRATION POINT)
  portfolioLink?: {
    type: 'application' | 'project';
    id: string; // Application or Project ID from Portfolio
    syncStatus: 'synced' | 'out-of-sync' | 'manual';
    lastSynced: string;
  };
  
  // Ownership
  owner: {
    id: string;
    name: string;
    email: string;
    role: string;
  };
  sponsor: {
    name: string;
    role: string;
  };
  team: TeamMember[];
  
  // Status
  status: 'not-started' | 'in-progress' | 'on-hold' | 'completed' | 'cancelled';
  currentStage: string; // Stage ID
  overallHealth: 'on-track' | 'at-risk' | 'critical';
  
  // Timeline
  plannedStartDate: string;
  plannedEndDate: string;
  actualStartDate?: string;
  forecastEndDate: string;
  actualEndDate?: string;
  
  // Progress
  stageProgress: StageProgress[];
  overallProgress: number; // 0-100
  
  // Gates
  gatesCompleted: number;
  totalGates: number;
  pendingGate?: {
    gateId: string;
    gateName: string;
    submittedDate: string;
    pendingApprovals: PendingApproval[];
  };
  
  // Documents & Deliverables
  deliverables: Deliverable[];
  documents: Document[];
  
  // Risks & Issues
  risks: Risk[];
  issues: Issue[];
  
  // Financial
  budget?: {
    total: number;
    spent: number;
    forecast: number;
  };
  
  // Benefits (especially for retirement/modernization)
  expectedBenefits: Benefit[];
  realizedBenefits: Benefit[];
  
  // Activity Log
  activityLog: ActivityLogEntry[];
  
  // Tags
  tags: string[];
}

interface TeamMember {
  id: string;
  name: string;
  email: string;
  role: string;
  allocation: number; // Percentage
}

interface StageProgress {
  stageId: string;
  stageName: string;
  status: 'not-started' | 'in-progress' | 'gate-pending' | 'completed';
  progress: number; // 0-100
  startDate?: string;
  endDate?: string;
  health: 'on-track' | 'at-risk' | 'delayed';
  completedActivities: number;
  totalActivities: number;
}

interface PendingApproval {
  approverName: string;
  approverRole: string;
  requestedDate: string;
  dueDate: string;
  status: 'pending' | 'approved' | 'rejected' | 'conditional';
  comments?: string;
  decisionDate?: string;
}

interface Deliverable {
  id: string;
  name: string;
  stage: string;
  dueDate: string;
  status: 'not-started' | 'in-progress' | 'completed' | 'overdue';
  owner: string;
  fileUrl?: string;
  completionDate?: string;
}

interface Document {
  id: string;
  title: string;
  type: string;
  uploadedBy: string;
  uploadedDate: string;
  fileUrl: string;
  fileSize: string;
}

interface Risk {
  id: string;
  description: string;
  category: 'technical' | 'resource' | 'schedule' | 'budget' | 'stakeholder';
  probability: 'low' | 'medium' | 'high';
  impact: 'low' | 'medium' | 'high';
  mitigation: string;
  owner: string;
  status: 'open' | 'mitigated' | 'closed';
}

interface Issue {
  id: string;
  description: string;
  severity: 'low' | 'medium' | 'high' | 'critical';
  raisedBy: string;
  raisedDate: string;
  status: 'open' | 'in-progress' | 'resolved' | 'closed';
  assignedTo: string;
  resolutionDate?: string;
}

interface Benefit {
  type: 'cost-savings' | 'efficiency' | 'revenue' | 'risk-reduction' | 'quality';
  description: string;
  quantifiedValue?: number;
  realizationDate: string;
  status: 'planned' | 'in-progress' | 'realized' | 'not-realized';
  actualValue?: number;
}

interface ActivityLogEntry {
  id: string;
  timestamp: string;
  actor: string;
  action: 'stage-completed' | 'gate-submitted' | 'approval-received' | 'document-uploaded' | 'risk-added' | 'status-changed';
  description: string;
  details?: any;
}

// Approval Workflow
interface ApprovalWorkflow {
  id: string;
  lifecycleInstanceId: string;
  lifecycleInstanceName: string;
  gate: {
    id: string;
    name: string;
    stage: string;
  };
  
  // Submission
  submittedBy: string;
  submittedDate: string;
  submissionNotes: string;
  
  // Status
  status: 'pending' | 'approved' | 'rejected' | 'conditional' | 'withdrawn';
  
  // Approvals
  approvals: ApprovalDecision[];
  requiredApprovals: number;
  receivedApprovals: number;
  
  // Documents
  supportingDocuments: string[]; // Document IDs
  
  // Timeline
  expectedDecisionDate: string;
  actualDecisionDate?: string;
  
  // Escalation
  escalated: boolean;
  escalationDate?: string;
  escalationReason?: string;
}

interface ApprovalDecision {
  approverName: string;
  approverRole: string;
  decision: 'approved' | 'rejected' | 'conditional';
  decisionDate: string;
  comments: string;
  conditions?: string[]; // If conditional approval
}

// Retirement Process (Special type closely linked to Portfolio)
interface RetirementProcess extends LifecycleInstance {
  type: 'application-retirement';
  
  // Portfolio Integration (CRITICAL)
  applicationId: string; // Direct link to Portfolio application
  applicationName: string;
  applicationCurrentStatus: string; // From Portfolio
  
  // Retirement Specifics
  retirementReason: 'end-of-life' | 'redundant' | 'consolidation' | 'cost-optimization' | 'security';
  replacementSystem?: string;
  
  // Migration
  dataMigration: {
    required: boolean;
    volumeGB?: number;
    targetSystem?: string;
    migrationPlan?: string;
    testingCompleted?: boolean;
  };
  
  // Users
  affectedUsers: number;
  userMigrationPlan?: string;
  trainingCompleted?: boolean;
  
  // Decommissioning
  infrastructureToDecommission: string[];
  licensesCancelled: string[];
  
  // Financial Impact
  annualCostSavings: number;
  decommissioningCost: number;
  netSavings: number;
  
  // Compliance
  dataRetentionPeriod: string;
  archiveLocation?: string;
  complianceSignOff?: {
    signedBy: string;
    signedDate: string;
  };
}
```
